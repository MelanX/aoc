\def\unless#1{#1\else\expandafter\expandafter\fi\iffalse\iftrue\fi}
\def\emptyline{}

\newcount\maxcalories
\newcount\curcalories

\maxcalories0\relax
\curcalories0\relax

\def\hasmore{
\ifnum\curcalories>\maxcalories\relax
\maxcalories\curcalories\relax
\fi
\curcalories0\relax
}

\newread\aocinput
\openin\aocinput=aocinput.txt
\loop\unless{\ifeof\aocinput}
\endlinechar-1\relax
\immediate\read\aocinput to \aoc
\ifx\aoc\emptyline\hasmore\else
\advance\curcalories\aoc\relax
\fi
\repeat
\closein\aocinput
\hasmore

\immediate\the\maxcalories

\bye
